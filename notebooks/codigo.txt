import networkx as nx
import matplotlib.pyplot as plt
import random

def visualizar_grafo(G, titulo="Grafo", pos=None):
    plt.figure(figsize=(6,5))

    if pos is None:
        pos = nx.spring_layout(G, seed= 42)

    nx.draw(
        G,
        pos,
        with_labels=True,
        node_size= 500
    )

    plt.title(f"{titulo}\nNodos: {G.number_of_nodes()} Aristas: {G.number_of_edges()}")
    plt.show()

#funciones para cada modelo.
def modelo_gnm(n,m):
    return nx.gnm_random_graph(n,m)

def modelo_gnp(n,p):
    return nx.gnp_random_graph(n,p)

def modelo_geografico(n,radio):
    return nx.random_geometric_graph(n,radio)

def modelo_dorogovtsev(generacion):
    return nx.dorogovtsev_goltsev_mendes_graph(generacion)

#Generación de un Grafo Gnm
n = random.randint(10,20)
m= random. randint(n, n+10)

G1= modelo_gnm(n,m)

visualizar_grafo(G1, f"Modelo Gnm (n={n}, m={m})")

#Generación de un grafo Gnp
n = random.randint(10,20)
p = random.uniform(0.2, 0.4)

G2 = modelo_gnp(n,p)
visualizar_grafo(G2, f"Modelo Gnp (n={n}, p={p:.2f})")  

#Geración de un grafo geográfico
n= random.randint(10,20)
radio= 0.35

G3= modelo_geografico(n,radio)

pos=nx.get_node_attributes(G3, 'pos')
visualizar_grafo(G3, f"Modelo Geográfico (n={n})", pos)

#Generación de un grafo Dorogovtsev-Mendes
generacion= 2

G4= modelo_dorogovtsev(generacion)

visualizar_grafo(G4, f"Modelo Dorogovtsev-Mendes (generación={generacion})")

def resumen(G):
    print("Nodos:", G.number_of_nodes())
    print("Aristas:", G.number_of_edges())
    print("Grado promedio:", sum(dict(G.degree()).values())/G.number_of_nodes())
    print()

resumen(G1)
resumen(G2)
resumen(G3)
resumen(G4)
